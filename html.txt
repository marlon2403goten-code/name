    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, doc, setDoc, addDoc, getDoc, onSnapshot, deleteDoc, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'motorbike-repuestos-v1';
        const apiKey = ""; 

        let productos = [];
        let cart = [];
        let currentCategory = 'all';
        let currentBrand = 'all';
        let isAdmin = false;
        let user = null;
        let currentImageBase64 = "";
        
        const _idp = atob("TWFybG9uMjQwMw=="); // Marlon2403 oculto
        const _dest = "3117718703";

        let currentAdminPass = "admin123"; 
        let activeVerificationCode = null;

        // --- LÓGICA GEMINI API ---
        async function askGemini(prompt, systemInstruction = "") {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: prompt }] }],
                systemInstruction: { parts: [{ text: systemInstruction }] }
            };

            for (let i = 0; i < 5; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (!response.ok) throw new Error('API Error');
                    const result = await response.json();
                    return result.candidates?.[0]?.content?.parts?.[0]?.text;
                } catch (err) {
                    await new Promise(r => setTimeout(r, Math.pow(2, i) * 1000));
                }
            }
            return "Lo siento, tuve un problema al conectarme con mi cerebro de IA. Intenta de nuevo.";
        }

        window.generateAIDescription = async () => {
            const name = document.getElementById('form-name').value;
            const ref = document.getElementById('form-ref').value;
            if(!name) return alert("Ingresa al menos el nombre del producto.");

            const btn = event.currentTarget;
            btn.innerText = "✨ Generando...";
            btn.disabled = true;

            const prompt = `Genera una descripción comercial y técnica breve (máximo 150 caracteres) para un repuesto de moto llamado "${name}" con referencia "${ref}". Sé profesional y persuasivo.`;
            const result = await askGemini(prompt, "Eres un experto en ventas de repuestos de motos para Motorbike Mayorista.");
            
            document.getElementById('form-desc').value = result;
            btn.innerText = "✨ Generar IA";
            btn.disabled = false;
        };

        window.sendAIMessage = async () => {
            const input = document.getElementById('ai-input');
            const container = document.getElementById('ai-messages');
            const msg = input.value.trim();
            if(!msg) return;

            input.value = "";
            container.innerHTML += `<div class="message-user p-3 rounded-2xl text-xs max-w-[85%]">${msg}</div>`;
            container.scrollTop = container.scrollHeight;

            const inventoryContext = productos.map(p => `- ${p.nombre} (Ref: ${p.referencia}, Precio: ${formatCOP(p.precioMayorista)})`).join("\n");
            const systemPrompt = `Eres el asistente inteligente de "Motorbike Mayorista" en Villavicencio. 
            Ayuda a los clientes a encontrar repuestos. No inventes productos. 
            Este es nuestro catálogo actual:\n${inventoryContext}\n
            Responde de forma amable, motera y técnica. Si no tenemos algo, sugiere lo más parecido o invita a contactar a Marlon vía WhatsApp.`;

            const aiResponse = await askGemini(msg, systemPrompt);
            container.innerHTML += `<div class="message-ai p-3 rounded-2xl text-xs max-w-[85%]">${aiResponse}</div>`;
            container.scrollTop = container.scrollHeight;
        };

        // --- FIREBASE INIT ---
        const initAuth = async () => {
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) await signInWithCustomToken(auth, __initial_auth_token);
                else await signInAnonymously(auth);
            } catch (e) { console.error("Auth Error:", e); }
        };
        initAuth();

        onAuthStateChanged(auth, async (u) => {
            user = u;
            if (user) {
                const q = collection(db, 'artifacts', appId, 'public', 'data', 'productos');
                onSnapshot(q, (snapshot) => { 
                    productos = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })); 
                    render(); 
                }, (err) => console.error("Firestore Error:", err));

                try {
                    const configDoc = await getDoc(doc(db, 'artifacts', appId, 'public', 'data', 'config', 'security'));
                    if (configDoc.exists()) currentAdminPass = configDoc.data().adminPass;
                } catch (e) {}
            }
        });

        // --- UI HELPERS ---
        window.formatCOP = (price) => '$' + Number(price).toLocaleString('es-CO');
        window.toggleCart = () => document.getElementById('cart-sidebar').classList.toggle('open');
        window.openAdmin = () => document.getElementById('admin-sidebar').classList.add('open');
        window.closeAdmin = () => { document.getElementById('admin-sidebar').classList.remove('open'); resetPassFlow(); };
        window.toggleAIChat = () => document.getElementById('ai-chat-sidebar').classList.toggle('open');

        window.toggleAdminBrandSelect = () => {
            const cat = document.getElementById('form-cat').value;
            const container = document.getElementById('admin-brand-select-container');
            if (cat === 'repuestos') container.classList.remove('hidden');
            else container.classList.add('hidden');
        };

        window.checkAdminAuth = () => {
            const passInput = document.getElementById('admin-pass').value;
            if (passInput === currentAdminPass) { 
                isAdmin = true; 
                document.getElementById('admin-login-ui').classList.add('hidden'); 
                document.getElementById('admin-controls-ui').classList.remove('hidden'); 
                render(); 
            }
            else document.getElementById('admin-error').classList.remove('hidden');
        };

        window.logoutAdmin = () => { 
            isAdmin = false; 
            document.getElementById('admin-login-ui').classList.remove('hidden'); 
            document.getElementById('admin-controls-ui').classList.add('hidden'); 
            document.getElementById('admin-pass').value = ''; 
            render(); 
        };

        window.requestPassCode = async () => {
            const masterInput = document.getElementById('master-key-input');
            const btn = document.getElementById('btn-request-code');

            if (!masterInput || masterInput.value.trim() !== _idp) {
                alert("Acceso Denegado");
                return;
            }

            try {
                btn.disabled = true;
                btn.innerText = "Validando...";
                activeVerificationCode = Math.floor(100000 + Math.random() * 900000).toString();
                const msg = `🛡️ SEGURIDAD MOTORBIKE 🛡️\n\nIdentidad validada.\n\nTu código privado de recuperación es: ${activeVerificationCode}\n\nIngresa este código en el panel para cambiar tu contraseña.`;
                const waUrl = `https://wa.me/57${_dest}?text=${encodeURIComponent(msg)}`;
                const win = window.open(waUrl, '_blank');
                if (!win || win.closed || typeof win.closed === 'undefined') window.location.assign(waUrl);
                document.getElementById('pass-request-step').classList.add('hidden');
                document.getElementById('pass-verify-step').classList.remove('hidden');
            } catch (e) {
                btn.disabled = false;
                btn.innerText = "Validar y Recibir Código";
            }
        };

        window.verifyAndChangePass = async () => {
            const codeInput = document.getElementById('verify-code-input').value.trim();
            const newPass = document.getElementById('new-pass-input').value.trim();
            if (!activeVerificationCode || codeInput !== activeVerificationCode) return alert("Código incorrecto.");
            if (newPass.length < 4) return alert("Contraseña muy corta.");
            try {
                await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'config', 'security'), { adminPass: newPass });
                currentAdminPass = newPass;
                alert("¡CONTRASEÑA ACTUALIZADA!");
                resetPassFlow();
            } catch (e) { alert("Error al guardar."); }
        };

        const resetPassFlow = () => {
            const r = document.getElementById('pass-request-step');
            const v = document.getElementById('pass-verify-step');
            if (r) r.classList.remove('hidden');
            if (v) v.classList.add('hidden');
            activeVerificationCode = null;
            const btn = document.getElementById('btn-request-code');
            if (btn) { btn.disabled = false; btn.innerText = "Validar y Recibir Código"; }
        };

        window.handleFileSelect = (event) => {
            const file = event.target.files[0];
            if (file && file.type.startsWith('image/')) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    currentImageBase64 = e.target.result;
                    const preview = document.getElementById('image-preview');
                    preview.src = currentImageBase64;
                    preview.style.display = 'block';
                    document.getElementById('drop-text').style.display = 'none';
                };
                reader.readAsDataURL(file);
            }
        };

        window.saveProduct = async () => {
            if (!user) return;
            const pId = document.getElementById('edit-id').value;
            const cat = document.getElementById('form-cat').value;
            const data = {
                nombre: document.getElementById('form-name').value.trim(),
                referencia: document.getElementById('form-ref').value.trim(),
                descripcion: document.getElementById('form-desc').value.trim(),
                precioDetal: Number(document.getElementById('form-detal').value),
                precioMayorista: Number(document.getElementById('form-mayor').value),
                categoria: cat,
                marca: cat === 'repuestos' ? document.getElementById('form-brand').value : 'GENERICO',
                imagen: currentImageBase64 || "https://images.unsplash.com/photo-1558981403-c5f9899a28bc?q=80&w=400"
            };
            if (!data.nombre || !data.referencia) return;
            try {
                const docRef = pId ? doc(db, 'artifacts', appId, 'public', 'data', 'productos', pId) : doc(collection(db, 'artifacts', appId, 'public', 'data', 'productos'));
                await setDoc(docRef, data);
                resetForm();
                const msgArea = document.getElementById('admin-msg');
                msgArea.innerText = "¡Guardado!";
                setTimeout(() => msgArea.innerText = "", 2000);
            } catch (e) { console.error(e); }
        };

        window.editProduct = (id) => {
            const p = productos.find(x => x.id === id);
            if (p) {
                document.getElementById('edit-id').value = p.id;
                document.getElementById('form-name').value = p.nombre;
                document.getElementById('form-ref').value = p.referencia;
                document.getElementById('form-desc').value = p.descripcion || "";
                document.getElementById('form-detal').value = p.precioDetal;
                document.getElementById('form-mayor').value = p.precioMayorista;
                document.getElementById('form-cat').value = p.categoria;
                
                if (p.categoria === 'repuestos') {
                    document.getElementById('admin-brand-select-container').classList.remove('hidden');
                    document.getElementById('form-brand').value = p.marca || 'GENERICO';
                } else {
                    document.getElementById('admin-brand-select-container').classList.add('hidden');
                }

                currentImageBase64 = p.imagen;
                const preview = document.getElementById('image-preview');
                preview.src = p.imagen;
                preview.style.display = 'block';
                document.getElementById('drop-text').style.display = 'none';
                openAdmin();
            }
        };

        window.deleteProduct = async (id) => { 
            if (confirm("¿Eliminar permanentemente?")) await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'productos', id)); 
        };

        window.resetForm = () => {
            document.getElementById('edit-id').value = '';
            document.getElementById('form-name').value = '';
            document.getElementById('form-ref').value = '';
            document.getElementById('form-desc').value = '';
            document.getElementById('form-detal').value = '';
            document.getElementById('form-mayor').value = '';
            document.getElementById('admin-brand-select-container').classList.add('hidden');
            currentImageBase64 = "";
            document.getElementById('image-preview').style.display = 'none';
            document.getElementById('drop-text').style.display = 'block';
        };

        window.addToCart = (id) => {
            const p = productos.find(x => x.id === id); if (!p) return;
            const inCart = cart.find(item => item.id === id);
            if (inCart) inCart.qty++; else cart.push({ ...p, qty: 1 });
            updateCartUI(); toggleCart();
        };

        window.updateCartQty = (id, delta) => {
            const i = cart.find(x => x.id === id);
            if (i) { i.qty += delta; if (i.qty <= 0) cart = cart.filter(x => x.id !== id); updateCartUI(); }
        };

        const updateCartUI = () => {
            const c = document.getElementById('cart-items');
            const empty = document.getElementById('empty-cart-msg');
            if (!c) return; c.innerHTML = ''; let tg = 0, ti = 0;
            if (cart.length === 0) { if (empty) empty.classList.remove('hidden'); } else {
                if (empty) empty.classList.add('hidden');
                cart.forEach(item => {
                    const st = item.precioMayorista * item.qty; tg += st; ti += item.qty;
                    c.innerHTML += `
                        <div class="flex items-center gap-4 bg-white/5 p-4 rounded-xl border border-white/5">
                            <img src="${item.imagen}" class="w-14 h-14 rounded-lg object-cover">
                            <div class="flex-grow">
                                <h4 class="text-[10px] font-black text-white uppercase line-clamp-1">${item.nombre}</h4>
                                <div class="flex items-center justify-between">
                                    <div class="flex items-center gap-3">
                                        <button onclick="updateCartQty('${item.id}', -1)" class="w-6 h-6 rounded bg-white/10 hover:bg-neon hover:text-black">-</button>
                                        <span class="text-xs font-bold text-white">${item.qty}</span>
                                        <button onclick="updateCartQty('${item.id}', 1)" class="w-6 h-6 rounded bg-white/10 hover:bg-neon hover:text-black">+</button>
                                    </div>
                                    <p class="text-xs font-black text-neon">${formatCOP(st)}</p>
                                </div>
                            </div>
                        </div>`;
                });
            }
            document.getElementById('cart-count').innerText = ti;
            document.getElementById('cart-total').innerText = formatCOP(tg);
        };

        window.sendWhatsAppOrder = () => {
            if (cart.length === 0) return;
            let msg = "🏍️ *PEDIDO MOTORBIKE*\n\nHola, solicito los siguientes productos:\n\n";
            cart.forEach(i => msg += `• ${i.nombre}\n  REF: ${i.referencia}\n  Cant: ${i.qty} x ${formatCOP(i.precioMayorista)}\n\n`);
            msg += `💰 *TOTAL: ${formatCOP(cart.reduce((a, b) => a + (b.precioMayorista * b.qty), 0))}*`;
            window.open(`https://wa.me/573118799249?text=${encodeURIComponent(msg)}`, '_blank');
        };

        const render = () => {
            const grid = document.getElementById('product-grid');
            const no = document.getElementById('no-results-message');
            const term = document.getElementById('searchInput') ? document.getElementById('searchInput').value.toLowerCase() : '';
            if (!grid) return; grid.innerHTML = '';
            
            const filtered = productos.filter(p => {
                const matchCat = (currentCategory === 'all' || p.categoria === currentCategory);
                const matchBrand = (currentCategory !== 'repuestos' || currentBrand === 'all' || p.marca === currentBrand);
                const matchSearch = (p.nombre.toLowerCase().includes(term) || p.referencia.toLowerCase().includes(term));
                return matchCat && matchBrand && matchSearch;
            });

            if (filtered.length === 0) { if (no) no.classList.remove('hidden'); return; }
            if (no) no.classList.add('hidden');
            
            filtered.forEach(p => {
                grid.innerHTML += `
                    <div class="neon-border-hover bg-white/5 border border-white/10 rounded-2xl overflow-hidden group relative">
                        ${isAdmin ? `<div class="absolute top-2 right-2 z-20 flex gap-1"><button onclick="editProduct('${p.id}')" class="w-8 h-8 bg-blue-600 rounded-lg flex items-center justify-center text-[10px] shadow-lg"><i class="fas fa-edit"></i></button><button onclick="deleteProduct('${p.id}')" class="w-8 h-8 bg-red-600 rounded-lg flex items-center justify-center text-[10px] shadow-lg"><i class="fas fa-trash"></i></button></div>` : ''}
                        <div class="h-48 overflow-hidden relative">
                            <div class="absolute top-4 left-4 z-10 bg-black/80 px-2 py-0.5 rounded text-[7px] font-black uppercase text-neon border border-neon/20">${p.categoria} ${p.marca && p.marca !== 'GENERICO' ? '• ' + p.marca : ''}</div>
                            <img src="${p.imagen}" class="w-full h-full object-cover grayscale-[30%] group-hover:grayscale-0 transition duration-500">
                        </div>
                        <div class="p-6">
                            <h3 class="font-bold text-xs mb-1 line-clamp-2 min-h-[32px] text-white uppercase">${p.nombre}</h3>
                            <p class="text-[9px] text-slate-500 mb-2">REF: ${p.referencia}</p>
                            <p class="text-[8px] text-slate-400 mb-4 line-clamp-2 italic">${p.descripcion || ''}</p>
                            <div class="bg-black/40 p-3 rounded-xl mb-4 border-l-2 border-neon">
                                <div class="flex justify-between items-center text-[9px] text-slate-500 mb-1"><span>P. Sugerido:</span><span class="line-through">${formatCOP(p.precioDetal)}</span></div>
                                <div class="flex justify-between items-center text-white"><span class="text-lg font-brand text-neon">${formatCOP(p.precioMayorista)}</span></div>
                            </div>
                            <button onclick="addToCart('${p.id}')" class="w-full bg-white/5 hover:bg-neon hover:text-black text-[10px] font-black py-3 rounded-lg transition-all border border-white/10 uppercase">Agregar al Pedido</button>
                        </div>
                    </div>`;
            });
        };

        window.filter = (c) => { 
            currentCategory = c; 
            currentBrand = 'all'; // Reset brand on category change
            
            const brandBar = document.getElementById('brands-filter-bar');
            if (c === 'repuestos') brandBar.classList.remove('hidden');
            else brandBar.classList.add('hidden');

            document.querySelectorAll('.filter-pill').forEach(b => b.classList.remove('active')); 
            if (event && event.target) event.target.classList.add('active'); 
            
            // Reset active brand pill
            document.querySelectorAll('.brand-pill').forEach(b => b.classList.remove('active'));
            document.querySelector('.brand-pill[onclick*="all"]').classList.add('active');

            render(); 
        };

        window.filterBrand = (brand) => {
            currentBrand = brand;
            document.querySelectorAll('.brand-pill').forEach(b => b.classList.remove('active'));
            if (event && event.target) event.target.classList.add('active');
            render();
        };

        const sInput = document.getElementById('searchInput');
        if (sInput) sInput.addEventListener('input', render);
    </script>
